• init from a multiple select

    					var selector1 = new MooMultiSelector({
						container: 'con1',
						local: true, //default value
						prefix: 'one', 
						from: 'tech', 
						size: {
							width: 302,
							height: 32
						},
						onSelect: function(item) {
							console.dir(item);
						},
						onUnselect: function(item) {
							console.dir(item);
						}
					});

					$('trans2Select').addEvent('click', function() {
						selector1.trans2Select();
					});

  • init from 'data' option

    					var selector2 = new MooMultiSelector({
						container: 'con2',
						local: true, //default value
						prefix: 'two',
						size: {
							width: 302,
							height: 32
						},
						data: [
							{"value": "javascript"},
							{"value": "nodejs", "selected": true},
							{"value": "jQuery", "selected": true},
							{"value": "mootools", "label": "mootools", "selected": true},
							{"value": "css3", "label": "css3" },
							{"value": "html5", "label": "html5"},
							{"value": "json", "selected": true}
						]
					});

					$('get').addEvent('click', function() {
						console.dir(selector2.getSelected());
					});

  • init with ajax

    					var selector3 = new MooMultiSelector({
						container: 'con3',
						local: false, 
						prefix: 'three',
						load: 'after',  
						size: {
							width: 302,
							height: 32
						},
						remote: {
							url: 'data.json',
							secure: true,
							onSuccess: function() {
								console.log('load complete!');
							}
						}
					});

					$('load').addEvent('click', function() {
						selector3.load();
					});
